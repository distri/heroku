    If our request is absolute we use that url, otherwise we get the base url from our root and append the path. This allows us to follow HATEOS resource urls more easily.

    {get, api} = ApiGenerator (options) ->
  options = extend
    dataType: "json"
    headers:
      Accept: "application/vnd.heroku+json; version=3"
      Authorization: herokuKey
    type: "GET"
  , options

  path = options.url
  unless path.match /^http/
    options.url = "#{root}/#{path}"

  $.ajax(options)
